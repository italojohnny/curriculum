\documentclass[paper=a4,fontsize=12pt]{scrartcl} % KOMA-article class
	\usepackage[brazil]{babel}
	\usepackage[utf8x]{inputenc}
	\usepackage[protrusion=true,expansion=true]{microtype}
	\usepackage{amsmath,amsfonts,amsthm}     % Math packages
	\usepackage{graphicx}                    % Enable pdflatex
	\usepackage[svgnames]{xcolor}            % Colors by their 'svgnames'
	\usepackage{geometry}
		\textheight=700px                    % Saving trees ;-)
	\usepackage{url}

	\frenchspacing              % Better looking spacings after periods
	\pagestyle{empty}           % No pagenumbers/headers/footers
	\usepackage{float}
	\usepackage{wrapfig}

	%%% Custom sectioning (sectsty package)
	%%% ------------------------------------------------------------
	\usepackage{sectsty}

	\sectionfont{%			            % Change font of \section command
		\usefont{OT1}{phv}{b}{n}%		% bch-b-n: CharterBT-Bold font
		\sectionrule{0pt}{0pt}{-5pt}{3pt}}


	%%% Macros
	%%% ------------------------------------------------------------
	\newlength{\spacebox}
	\settowidth{\spacebox}{8888888888}			% Box to align text
	\newcommand{\sepspace}{\vspace*{1em}}		% Vertical space macro

	\newcommand{\MyName}[1]{ % Name
			\Huge \usefont{OT1}{phv}{b}{n} \hfill #1
			\par \normalsize \normalfont}

	\newcommand{\MySlogan}[1]{ % Slogan (optional)
			\large \usefont{OT1}{phv}{m}{n}\hfill \textit{#1}
			\par \normalsize \normalfont}

	\newcommand{\NewPart}[1]{\section*{\uppercase{#1}}}

	\newcommand{\PersonalEntry}[2]{
			\noindent\hangindent=2em\hangafter=0 % Indentation
			\parbox{\spacebox}{        % Box to align text
			\textit{#1}}		       % Entry name (birth, address, etc.)
			\hspace{1.5em} #2 \par}    % Entry value

	\newcommand{\SkillsEntry}[2]{      % Same as \PersonalEntry
			\noindent\hangindent=2em\hangafter=0 % Indentation
			\parbox{\spacebox}{        % Box to align text
			\textit{#1}}			   % Entry name (birth, address, etc.)
			\hspace{1.5em} #2 \par}    % Entry value
			%\hspace{1.5em}\texttt{#2} \par}    % Entry value

	\newcommand{\EducationEntry}[4]{
			\noindent \textbf{#1} \hfill      % Study
			\colorbox{Black}{%
				\parbox{8em}{%
				\hfill\color{White}
				#2
				}} \par  % Duration
			\noindent \textit{#3} \par        % School
			\noindent\hangindent=2em\hangafter=0 \small #4 % Description
			\normalsize \par}

	\newcommand{\WorkEntry}[4]{				  % Same as \EducationEntry
			\noindent \textbf{#1} \hfill      % Jobname
			\colorbox{Black}{\color{White}#2} \par  % Duration
			\noindent \textit{#3} \par              % Company
			\noindent\hangindent=2em\hangafter=0 \small #4 % Description
			\normalsize \par}


\begin{document}

	\MyName{< print detail.first-name.value >}
	\MySlogan{Curriculum Vitae}
	\sepspace


	\NewPart{< print detail.label >}{}
	\PersonalEntry{< print detail.birth.label >}{< print detail.birth.value >}
	\PersonalEntry{< print detail.city.label >}{< print detail.city.value >}
	\PersonalEntry{< print detail.phone.label >}{< print detail.phone.value >}
	\PersonalEntry{< print detail.email.label >}{\url{< print detail.email.value >}}
	\PersonalEntry{< print detail.linkedin.label >}{\url{< print detail.linkedin.value >}}
	\PersonalEntry{< print detail.github.label >}{\url{< print detail.github.value >}}


	\NewPart{< print experiences.label >}{}
	< for xp in experiences.list >
		\EducationEntry{< print xp.occupation.value >}
		{< print xp.start-date.value > - < print xp.closing-date.value >\space\space\space\space}
		{< print xp.company.value >., < print xp.period.value >}
		{< print xp.description.value >}
		\sepspace
	< endfor >
	\newpage


	\NewPart{< print educations.label >}{}
	< for ed in educations.list >
		\EducationEntry{< print ed.title.value >}{< print ed.start-date.value > - < print ed.closing-date.value > \space\space\space\space\space}
		{< print ed.institute.value >}
		{< print ed.description.value >}
		\sepspace
	< endfor >


	\NewPart{< print skills.label >}{}
	< for sk in skills.list >
		\SkillsEntry{< print sk.name.value >}{< print sk.description.value >}
		\SkillsEntry{}{}
	< endfor >

\end{document}
